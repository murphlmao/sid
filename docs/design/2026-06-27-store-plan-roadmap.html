2 · The contract every tab touches (illustrative)

enum Scope { Global, Workspace(WorkspaceId) }
struct Attributed<T> { item: T, origin: Scope, duplicate: bool }   // provenance + dup flag

trait Store {
    fn read<T>(&self, scope: Scope, filters: ViewFilters) -> Result<Vec<Attributed<T>>>;  // composed union
    fn write<T>(&self, item: T, scope: Scope) -> Result<()>;                       // save to layer
    fn promote<T>(&self, id: &str, from: Scope) -> Result<()>;  // ws → global
    fn demote<T>(&self, id: &str, to: WorkspaceId) -> Result<()>; // global → ws
}
// ViewFilters { collapse_duplicates: bool = true, hide_global: bool = false }  — your two checkboxes
